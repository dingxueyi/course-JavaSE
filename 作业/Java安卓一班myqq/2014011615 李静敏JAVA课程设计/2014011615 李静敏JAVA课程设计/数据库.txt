CREATE TABLE `users` (
  `qqNum` INTEGER(11) NOT NULL AUTO_INCREMENT,
  `nickNames` VARCHAR(20) COLLATE utf8_general_ci NOT NULL,
  `password` VARCHAR(20) COLLATE utf8_general_ci NOT NULL,
  `registTime` DATETIME(6) DEFAULT NULL,
  `gender` CHAR(2) COLLATE utf8_general_ci NOT NULL,
  `introduce` VARCHAR(40) COLLATE utf8_general_ci DEFAULT NULL,
  `ip` VARCHAR(20) COLLATE utf8_general_ci DEFAULT NULL,
  PRIMARY KEY (`qqNum`) USING BTREE
) ENGINE=InnoDB
AUTO_INCREMENT=326 CHARACTER SET 'utf8' COLLATE 'utf8_general_ci'
;
CREATE TABLE `message` (
  `id` INTEGER(11) NOT NULL AUTO_INCREMENT,
  `sender` INTEGER(11) NOT NULL,
  `content` VARCHAR(40) COLLATE utf8_general_ci NOT NULL,
  `receiver` INTEGER(11) NOT NULL,
  `sendTime` DATETIME(6) DEFAULT NULL,
  `recieveTime` DATETIME(6) NOT NULL DEFAULT '0000-00-00 00:00:00.000000',
  `state` INTEGER(11) NOT NULL,
  PRIMARY KEY (`id`) USING BTREE
) ENGINE=InnoDB
AUTO_INCREMENT=47 CHARACTER SET 'utf8' COLLATE 'utf8_general_ci'
;
